function forzareRisultato() { let totalCentiseconds = Math.floor(elapsedTime / 10); let realSeconds = Math.floor(totalCentiseconds / 100) % 60; let realMinutes = Math.floor(Math.floor(totalCentiseconds / 100) / 60); let minStr = realMinutes < 10 ? '0' + realMinutes : realMinutes.toString(); let secStr = realSeconds < 10 ? '0' + realSeconds : realSeconds.toString(); // Scompone e somma le singole cifre di minuti e secondi reali let cifreTempo = minStr.split('').concat(secStr.split('')); let sommaMinutiESecondi = cifreTempo.reduce((sum, digit) => sum + parseInt(digit), 0); // Calcola quanto devono sommare le due cifre dei centesimi let diffPerCentesimi = targetSum - sommaMinutiESecondi; let forcedCentiseconds = 0; let realCent = totalCentiseconds % 100; let migliorDistanza = 999; let trovato = false; // Cerca la coppia di cifre (00-99) che soddisfa la differenza for (let c1 = 0; c1 <= 9; c1++) { for (let c2 = 0; c2 <= 9; c2++) { if (c1 + c2 === diffPerCentesimi) { let valoreCent = c1 * 10 + c2; let distanza = Math.abs(valoreCent - realCent); // Prende i centesimi più vicini a quelli reali dello stop if (distanza < migliorDistanza) { migliorDistanza = distanza; forcedCentiseconds = valoreCent; trovato = true; } } } } // Se per qualche motivo assurdo non trova la combinazione, lascia i centesimi reali if (!trovato) { forcedCentiseconds = realCent; } // Formatta i centesimi forzati a due cifre (es. 5 -> "05") let centStr = forcedCentiseconds < 10 ? '0' + forcedCentiseconds : forcedCentiseconds.toString(); // Aggiorna il display del vero cronometro Apple con il tempo truccato display.textContent = `${minStr}:${secStr},${centStr}`; }
